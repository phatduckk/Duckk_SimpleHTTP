  477.      * You can get to that data using $this->getResponseBody() and $this->getResponseHeaders()

  481.     public function read()
  482.     {
  483.         // read the header
  484.         $rawHeaders = '';
  485.         while ($b = fgets($this->socket, 8096)) {
  486.             $rawHeaders .= $b;
  487.             if (trim($b) == '') {
  488.                 $rawHeaders = rtrim($rawHeaders);
  489.                 break;
  490.             }
  491.         }
  492.  
  493.         $this->rawResponse   = "{$rawHeaders}\r\n\r\n";
  494.         $headerParts         = explode("\r\n", $rawHeaders);
  495.         $contentLengthKey    = null;
  496.         $transferEncodingKey = null;
  497.         $connectionKey       = null;
  498.  
  499.         if (! empty($headerParts)) {
  500.             foreach ($headerParts as $header) {
  501.                 if (preg_match('(^HTTP/(?P<version>\d+\.\d+)\s+(?P<status>\d+))S', $header, $matches)){
  502.                     $this->respStatus  = (int) $matches['status'];
  503.                 } else {
  504.                     if (! strstr($header, ':')) {
  505.                         // TODO: parsing the response is probably fucked up if this happens
  506.                         throw new Duckk_SimpleHTTP_Exception(
  507.                             'Invalid header. Odds are it was a bad request.'
  508.                         );
  509.                     }
  510.  
  511.                     list($headerName, $headerValue) = explode(':', $header, 2);
  512.                     $this->respHeaders[$headerName] = trim($headerValue);
  513.                     $lcHeaderName                   = strtolower($headerName);
  514.  
  515.                     if ($lcHeaderName == 'content-length') {
  516.                         $contentLengthKey = $headerName;
  517.                     } else if ($lcHeaderName == 'transfer-encoding') {
