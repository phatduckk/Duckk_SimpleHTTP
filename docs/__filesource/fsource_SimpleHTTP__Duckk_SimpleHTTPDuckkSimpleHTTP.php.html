  484.             foreach ($headerParts as $header) {
  485.                 if (preg_match('(^HTTP/(?P<version>\d+\.\d+)\s+(?P<status>\d+))S', $header, $matches)){
  486.                     $this->respStatus  = (int) $matches['status'];
  487.                 } else {
  488.                     if (! strstr($header, ':')) {
  489.                         // TODO: parsing the response is probably fucked up if this happens
  490.                         throw new Duckk_SimpleHTTP_Exception(
  491.                             'Invalid header. Odds are it was a bad request.'
  492.                         );
  493.                     }
  494.  
  495.                     list($headerName, $headerValue) = explode(':', $header, 2);
  496.                     $this->respHeaders[$headerName] = trim($headerValue);

  498.  
  499.                     if ($lcHeaderName == 'content-length') {
  500.                         $contentLengthKey = $headerName;
  501.                     } else if ($lcHeaderName == 'transfer-encoding') {
  502.                         $transferEncodingKey = $headerName;
  503.                     } else if ($lcHeaderName == 'connection') {
  504.                         $connectionKey = $headerName;
  505.                     }
  506.                 }
  507.             }
  508.  
  509.             $this->debug($this->respHeaders); // debug response headers
